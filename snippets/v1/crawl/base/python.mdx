```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
  api_key="fc-YOUR_API_KEY"
)

# Crawl a website
crawl_result = app.crawl_url(
  'https://firecrawl.dev',
  params={
    'limit': 100,
    'scrapeOptions': {
      'formats': ['markdown', 'html']
    }
  },
  poll_interval=30
)

print(crawl_result)
```
