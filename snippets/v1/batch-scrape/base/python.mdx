```python Python
from firecrawl import FirecrawlApp

app = FirecrawlApp(
  api_key='fc-YOUR_API_KEY'
)

# Scrape multiple websites synchronously
batch_scrape_response = app.batch_scrape_urls(
    urls=['https://firecrawl.dev', 'https://mendable.ai/'],
    params={
        'formats': ['markdown', 'html']
    }
)

print(batch_scrape_response)

# Scrape multiple websites asynchronously
batch_scrape_job = app.async_batch_scrape_urls(
    urls=['https://firecrawl.dev', 'https://mendable.ai/'],
    params={
        'formats': ['markdown', 'html']
    }
)

print(batch_scrape_job)

# When scraping asynchronously, you can then use the job ID to check the status of the batch scrape
batch_scrape_status = app.check_batch_scrape_status(
    id=batch_scrape_job['id']
)

print(batch_scrape_status)
```
