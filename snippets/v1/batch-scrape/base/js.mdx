```js Node
import FirecrawlApp from '@mendable/firecrawl-js';

const app = new FirecrawlApp({
  apiKey: 'fc-YOUR_API_KEY'
});

async function main() {
  // Scrape multiple websites synchronously
  const batchScrapeResponse = await app.batchScrapeUrls(
    ['https://firecrawl.dev', 'https://mendable.ai/'],
    {
      formats: ['markdown', 'html'],
    }
  );

  console.log(batchScrapeResponse);

  // Scrape multiple websites asynchronously
  const batchScrapeJob = await app.asyncBatchScrapeUrls(
    ['https://firecrawl.dev', 'https://mendable.ai/'],
    {
      formats: ['markdown', 'html'],
    }
  );

  console.log(batchScrapeJob);

  // When scraping asynchronously, you can then use the job ID to check the status of the batch scrape
  const batchScrapeStatus = await app.checkBatchScrapeStatus(batchScrapeJob.id);

  console.log(batchScrapeStatus);
}

main();
```
