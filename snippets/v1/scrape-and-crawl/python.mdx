```python Python
# Import the Firecrawl SDK 
from firecrawl import FirecrawlApp

# Initialize Firecrawl with your API key
app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# Basic example: Scrape a single webpage
# This gets both markdown and HTML formats of the page content
scrape_result = app.scrape_url(
    'https://firecrawl.dev',
    params={
        # Choose which formats you want returned
        'formats': ['markdown', 'html']
    }
)

# Print the scraped content
print("Scraped page content:", scrape_result)

# Advanced example: Crawl multiple pages of a website
# This crawls up to 100 pages and gets markdown/HTML for each
crawl_result = app.crawl_url(
    'https://firecrawl.dev',
    params={
        # Limit how many pages to crawl
        'limit': 100,
        
        # Configure how to scrape each page
        'scrapeOptions': {
            'formats': ['markdown', 'html']
        }
    }
)

# Print the crawled content
print("Crawled website content:", crawl_result)
```
