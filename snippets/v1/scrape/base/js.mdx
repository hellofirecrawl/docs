```js Node
import FirecrawlApp from '@mendable/firecrawl-js';

const app = new FirecrawlApp({
  apiKey: 'fc-YOUR_API_KEY'
});

async function main() {
  // Scrape a website
  const scrapeResponse = await app.scrapeUrl(
    'https://firecrawl.dev',
    {
      formats: ['markdown', 'html'],
    }
  );

  console.log(scrapeResponse);
}

main();
```
